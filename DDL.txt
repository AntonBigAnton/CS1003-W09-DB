DROP TABLE IF EXISTS Persons;
CREATE TABLE Persons (pid VARCHAR(100) NOT NULL PRIMARY KEY, name VARCHAR(100) NOT NULL);
DROP TABLE IF EXISTS Venues;
CREATE TABLE Venues (venkey VARCHAR(100) NOT NULL PRIMARY KEY, title VARCHAR(100) NOT NULL);
DROP TABLE IF EXISTS Publications;
CREATE TABLE Publications (pubkey VARCHAR(100) NOT NULL PRIMARY KEY, pubyear INTEGER NOT NULL, title VARCHAR(100) NOT NULL, venkey VARCHAR(100), FOREIGN KEY (venkey) REFERENCES Venues(venkey));
DROP TABLE IF EXISTS Links;
CREATE TABLE Links (pid VARCHAR(100), pubkey VARCHAR(100), FOREIGN KEY (pid) REFERENCES Persons(pid), FOREIGN KEY (pubkey) REFERENCES Publications(pubkey));
